GERAÇÃO DA MATRIZ DO GLOBO E DATA INTEGRATION // ========================================== const globeRadius = 1.5; const earthRadiusKm = 6371; // Raio real da Terra para a escala matemática const numPoints = 9000; const geometry = new THREE.BufferGeometry(); const positions = new Float32Array(numPoints * 3); const colors = new Float32Array(numPoints * 3); const phi = Math.PI * (3 - Math.sqrt(5)); // Proporção áurea de Fibonacci // Banco de dados em memória RAM (O(1)) para lermos o país sem travar a engine const mapaPontosMemoria = new Map(); for (let i = 0; i < numPoints; i++) { const y = 1 - (i / (numPoints - 1)) * 2; const radiusAtY = Math.sqrt(1 - y * y); const theta = phi * i; const posX = Math.cos(theta) * radiusAtY * globeRadius; const posY = y * globeRadius; const posZ = Math.sin(theta) * radiusAtY * globeRadius; positions[i * 3] = posX; positions[i * 3 + 1] = posY; positions[i * 3 + 2] = posZ; // Cor base tática para o oceano (Azul/Cinza profundo de radar militar) colors[i * 3] = 0.1; // R colors[i * 3 + 1] = 0.15; // G colors[i * 3 + 2] = 0.2; // B // Conversão matemática ultra precisa de 3D para coordenadas geodésicas reais const lat = Math.asin(posY / globeRadius) * (180 / Math.PI); const lon = Math.atan2(-posX, posZ) * (180 / Math.PI); // Mapeia o ponto na estrutura aberta editável mapaPontosMemoria.set(i, { index: i, lat: lat, lon: lon, pais: "Oceano / Território Neutro" }); } geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3)); geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3)); const material = new THREE.PointsMaterial({ size: 0.015, vertexColors: true, transparent: true, opacity: 0.8 }); const globe = new THREE.Points(geometry, material); scene.add(globe); // Esfera invisível de colisão dedicada para garantir cálculo de clique suave sem delay de FPS const hitSphereGeom = new THREE.SphereGeometry(globeRadius, 32, 32); const hitSphereMat = new THREE.MeshBasicMaterial({ transparent: true, opacity: 0 }); const hitSphere = new THREE.Mesh(hitSphereGeom, hitSphereMat); scene.add(hitSphere); // ========================================== // 4. CARREGAMENTO DO BANCO DE FRONTEIRAS ESTÁTICO (GEOJSON) // ========================================== // Usamos uma base mundial leve via D3. O navegador roda isso uma única vez na inicialização d3.json('https://githubusercontent.com') .then(geoJson => { const colorAttribute = geometry.attributes.color; mapaPontosMemoria.forEach(ponto => { // O D3 varre as fronteiras e injeta o país direto no nosso array de memória for (let i = 0; i < geoJson.features.length; i++) { const feature = geoJson.features[i]; if (d3.geoContains(feature, [ponto.lon, ponto.lat])) { ponto.pais = feature.properties.ADMIN || feature.properties.name || "Desconhecido"; // Delineamento visual estático: Pinta a terra com a cor padrão da marca LKS (Verde Tático) colorAttribute.setXYZ(ponto.index, 0.0, 0.4, 0.3); break; } } }); colorAttribute.needsUpdate = true; document.querySelector('.system-status span:nth-child(2)').innerText = "STATUS: MAP READY"; }) .catch(err => { console.error("Erro no carregamento geodésico:", err); }); // ========================================== // 5. MOTOR DE TELEMETRIA, INTERAÇÃO E DISTÂNCIA REAL // ========================================== const raycaster = new THREE.Raycaster(); const mouse = new THREE.Vector2(); let currentRings = []; // Fórmula real de Haversine para cálculo de metragens e distâncias militares em KM function calcularDistanciaRealKM(lat1, lon1, lat2, lon2) { const toRad = Math.PI / 180; const dLat = (lat2 - lat1) * toRad; const dLon = (lon2 - lon1) * toRad; const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(lat1 * toRad) * Math.cos(lat2 * toRad) * Math.sin(dLon / 2) * Math.sin(dLon / 2); return earthRadiusKm * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))); } window.addEventListener('click', (event) => { // Ignora o clique se o usuário estiver interagindo com o painel lateral HTML if (event.target.closest('#vector-panel')) return; mouse.x = (event.clientX / window.innerWidth) * 2 - 1; mouse.y = -(event.clientY / window.innerHeight) * 2 + 1; raycaster.setFromCamera(mouse, camera); const intersects = raycaster.intersectObject(hitSphere); if (intersects.length > 0) { const point = intersects[0].point; // Ajusta a colisão 3D baseando-se na rotação viva da esfera const localPoint = hitSphere.worldToLocal(point.clone()); const latClique = Math.asin(localPoint.y / globeRadius) * (180 / Math.PI); const lonClique = Math.atan2(-localPoint.x, localPoint.z) * (180 / Math.PI); // Captura dinamicamente o alcance em KM do caça selecionado no seu menu lateral! const raioAtaqueKm = selectedVector.rangeKm; // Desenha o anel de projeção perfeitamente escalonado na tela criarAnelEscalonado(point, intersects[0].normal, raioAtaqueKm, selectedVector.color); // Busca O(1): Descobre em qual país clicou varrendo o mapa local super rápido let paisAlvo = "Oceano / Território Neutro"; for (let [i, pt] of mapaPontosMemoria) { // Uma aproximação geométrica simples para achar o país do clique no mapa indexado if (Math.abs(pt.lat - latClique) < 1.5 && Math.abs(pt.lon - lonClique) < 1.5) { if (pt.pais !== "Oceano / Território Neutro") { paisAlvo = pt.pais; break; } } } // Processamento de Brilho de Alvo Real na Malha de Pontos const colorAttribute = geometry.attributes.color; mapaPontosMemoria.forEach(ponto => { const distanciaKM = calcularDistanciaRealKM(latClique, lonClique, ponto.lat, ponto.lon); if (distanciaKM <= raioAtaqueKm) { // Se o ponto está dentro do raio real do caça, acende com a cor exclusiva dele (do seu AIRCRAFT_DATABASE) const c = new THREE.Color(selectedVector.color); colorAttribute.setXYZ(ponto.index, c.r, c.g, c.b); } else { // Fora do raio, reseta para a cor padrão do relevo estático if (ponto.pais !== "Oceano / Território Neutro") { colorAttribute.setXYZ(ponto.index, 0.0, 0.4, 0.3); // Terra comum } else { colorAttribute.setXYZ(ponto.index, 0.1, 0.15, 0.2); // Mar comum } } }); colorAttribute.needsUpdate = true; // Atualiza o seu painel HUD de Telemetria com os dados reais calculados document.getElementById('coords-display').innerHTML = `LAT: ${latClique.toFixed(4)}° | LON: ${lonClique.toFixed(4)}°
` + `RAIO OPERACIONAL: ${raioAtaqueKm} KM`; document.getElementById('target-display').innerText = `ALVO: ${paisAlvo.toUpperCase()}`; if(paisAlvo !== "Oceano / Território Neutro") { document.getElementById('target-display').style.color = "var(--secondary)"; } else { document.getElementById('target-display').style.color = "rgba(255,255,255,0.4)"; } } }); function criarAnelEscalonado(position, normal, raioKm, hexadecimalColor) { // Conversão matemática milimétrica: Transforma a metragem real em KM na escala física 3D do Three.js const raioVisualThreeJS = (raioKm / earthRadiusKm) * globeRadius; // Cria uma linha perfeitamente circular ao redor do alvo const ringGeom = new THREE.RingGeometry(raioVisualThreeJS - 0.015, raioVisualThreeJS, 64); const ringMat = new THREE.MeshBasicMaterial({ color: hexadecimalColor, side: THREE.DoubleSide, transparent: true, opacity: 0.85, depthTest: false }); const ring = new THREE.Mesh(ringGeom, ringMat); ring.position.copy(position); ring.quaternion.setFromUnitVectors(new THREE.Vector3(0, 0, 1), normal); scene.add(ring); currentRings.push(ring); // Animação militar sóbria (Apenas fade out estável para não distorcer a escala real em KM)new TWEEN.Tween(ringMat).to({ opacity: 0 }, 2500).onComplete(() => { scene.remove(ring); }).start();}// ==========================================// 6. RENDER LOOP E RESPONSIVIDADE// ==========================================function animate(time) {requestAnimationFrame(animate);TWEEN.update(time);controls.update();// Sincronização de rotação tática contínua do globo e da zona de colisãoif(controls.getAzimuthalAngle() !== undefined) {globe.rotation.y += 0.0004;hitSphere.rotation.y += 0.0004;}renderer.render(scene, camera);}animate();window.addEventListener('resize', () => {camera.aspect = window.innerWidth / window.innerHeight;camera.updateProjectionMatrix();renderer.setSize(window.innerWidth, window.innerHeight);});
